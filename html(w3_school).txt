        const modalcontainer = document.querySelector('.js-modalcontainer')

        function showbuytickets() {
            modal.classList.add('open')

        }

        function hidebuyticket() {
            modal.classList.remove('open')
        }
        modalcontainer.addEventListener('click', function(evennt) {
            evennt.stopPropagation()
        })

        var header = document.getElementById('header');
        mobileMenu = document.getElementById('mb-menu')
        mobileMenu.onclick = function() {
            var close = header.clientHeight === 47;
            if (close) {
                header.style.height = 'auto'
            } else {
                header.style.height = null
            }
        }
        var menuItems = document.querySelectorAll('#nav li a[href*="#"]')
        for (var i = 0; i < menuItems.length; i++) {
            var menuItem = menuItems[i];
            menuItem.onclick = function() {
                var parentMenu = this.nextElementSibling && menuItem.nextElementSibling.classList.contains('subnav');
                if (!parentMenu) {
                    header.style.height = null

                } else {
                    event.preventDefault();
                }
            }
